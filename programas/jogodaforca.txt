string letrasUsadas[26];
string palavra[8];
palavra[0] = "c";
palavra[1] = "o";
palavra[2] = "m";
palavra[3] = "p";
palavra[4] = "i";
palavra[5] = "l";
palavra[6] = "a";
palavra[7] = "r";

def exibirEstado(string letrasUsadas, string palavra, int tentativasRestantes) {
    string estado = "";
    int i;
    for (i = 0; i < 8; i = i + 1) {
        string letra;
        letra = palavra[i];
        int j;
        for (j = 0; j < 26; j = j + 1) {
            if (letra == letrasUsadas[j]) {
                estado = estado + letra;
            } else {
                estado = estado + "_";
            } endif
        }
    }
    print "Palavra: " + estado + "\n";
    print "Letras usadas: " + letrasUsadas + "\n";

    if (tentativasRestantes == 5) {
        print "Tentativas restantes: 5\n";
    } endif

    if (tentativasRestantes == 4) {
        print "Tentativas restantes: 4\n";
    } endif

    if (tentativasRestantes == 3) {
        print "Tentativas restantes: 3\n";
    } endif

    if (tentativasRestantes == 2) {
        print "Tentativas restantes: 2\n";
    } endif

    if (tentativasRestantes == 1) {
        print "Tentativas restantes: 1\n";
    } endif
}

def verificarVitoria(string palavra, string letrasUsadas) {
    int venceu;
    venceu = 0;
    int i;
    for (i = 0; i < 26; i = i + 1) {
        string letraUsada;
        letraUsada = letrasUsadas[i];
        int j;
        for (j = 0; j < 8; j = j + 1) {
            string letra;
            letra = palavra[j];
            if (letraUsada == letra) {
                venceu = venceu + 1;
                break;
            } endif
        }
    }
    if (venceu == 8) {
        return 1;
    } endif
    return 0;
}

def void jogoDaForca(string letrasUsadas) {
    string palavra[8];
    palavra[0] = "c";
    palavra[1] = "o";
    palavra[2] = "m";
    palavra[3] = "p";
    palavra[4] = "i";
    palavra[5] = "l";
    palavra[6] = "a";
    palavra[7] = "r";

    print "Pegue um papel e uma caneta para anotar as letras usadas! Eu não consigo fazer isso por você! A palavra possui 8 letras!";

    int tentativasRestantes;
    for (tentativasRestantes = 5; tentativasRestantes > 0; tentativasRestantes = tentativasRestantes - 1) {
        exibirEstado(letrasUsadas, palavra, tentativasRestantes);
        print "Adivinhe uma letra: ";
        string letra;
        read(letra);

        int j;
        for (j = 0; j < 26; j = j + 1) {
            if (letra == letrasUsadas[j]) {
                print "Você já usou essa letra.";
            } endif
        }

        int i;
        for (i = 0; i < 8; i = i + 1) {
            if (letra == palavra[i]) {
                if (i == 0) {
                    print "A letra se encontra na posicao 0";
                } endif

                if (i == 1) {
                    print "A letra se encontra na posicao 1";
                } endif

                if (i == 2) {
                    print "A letra se encontra na posicao 2";
                } endif

                if (i == 3) {
                    print "A letra se encontra na posicao 3";
                } endif

                if (i == 4) {
                    print "A letra se encontra na posicao 4";
                } endif

                if (i == 5) {
                    print "A letra se encontra na posicao 5";
                } endif

                if (i == 6) {
                    print "A letra se encontra na posicao 6";
                } endif

                if (i == 7) {
                    print "A letra se encontra na posicao 7";
                } endif

                if (verificarVitoria(palavra, letrasUsadas)) {
                    print "Parabéns! Você ganhou! A palavra era '" + palavra + "'.\n";
                    break;
                } endif
            } else {
                tentativasRestantes = tentativasRestantes - 1;
            } endif
        }

        if (tentativasRestantes == 0) {
            print "Você perdeu! A palavra era '" + palavra + "'.\n";
        } endif
    }
}

def main() {
    jogoDaForca();
}
